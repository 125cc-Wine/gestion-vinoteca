function renderAll() {
  const grid = document.getElementById('grid');
  wines.forEach((w, i) => {
    const card = document.createElement('div');
    card.className = 'card';
    card.dataset.title = w.title.toLowerCase();
    const priceHtml = w.price
      ? `<div class="price">${w.price}</div>`
      : `<div class="no-price">Sin precio</div>`;
    card.innerHTML = `
      <div class="qr-box" id="qr-${i}"></div>
      <h3>${w.title}</h3>
      ${priceHtml}
      <a href="${w.url}" target="_blank">Ver en web ↗</a>
    `;
    grid.appendChild(card);
    new QRCode(document.getElementById('qr-' + i), {
      text: w.url, width: 140, height: 140,
      colorDark: '#5a1a2a', colorLight: '#ffffff',
    });
  });
}
function filterWines() {
  const q = document.getElementById('search').value.toLowerCase().trim();
  const cards = document.querySelectorAll('.card');
  let visible = 0;
  cards.forEach(card => {
    const match = !q || card.dataset.title.includes(q);
    card.classList.toggle('hidden', !match);
    if (match) visible++;
  });
  document.getElementById('count').textContent = `Mostrando ${visible} vinos`;
}
renderAll();
